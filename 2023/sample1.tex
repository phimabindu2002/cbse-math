\begin{enumerate}

\item If $\alpha$ and $\beta$ are the zeroes of the quadratic polynomial $p(x)=x^2-ax-b$, then the value of $\alpha^2+\beta^s$ is :

\begin{enumerate}
\item $a^2-2b$
\item $a^2+2b$
\item $b^2-2a$
\item $b^2+2a$
\end{enumerate}

\item Assertion(A): The polynomial $p(x)=x^2+3x+3$ has two real zeroes.
Reason(R): A quadratic polynomial can have at most two real zeroes.


\item
\begin{enumerate}
\item If $4\cot^2 45^\circ - \sec^2 60^\circ + \sin^2 60^\circ + p = \frac{3}{4}$, then find the value of $p$.
\textbf{OR}
\item If $cosA+ cos^2A=1$,then find the value of $sin^2A+sin^4A$.
\end{enumerate}


\item Prove that:\\
$[\frac{1}{cos\theta}-cos\theta][\frac{1}{sin\theta}-sin\theta]$=$\frac{1}{tan\theta+cot\theta}$



\item The value of k for which the pair of equations $kx=y+2$ and $6x=2y+3$ has infinitely many solutions,
\begin{enumerate}
\item is$k=3$
\item does not exist
\item is $k=-3$
\item is $k=4$
\end{enumerate}


\item If $2tan A=3$, then the value of $\frac{4sinA+3cosA}{4sinA-3cosA}$ is
\begin{enumerate}
\item $\frac{7}{\sqrt{13}}$
\item $\frac{1}{\sqrt{13}}$
\item $3$
\item does not exist
\end{enumerate}


\item If $\alpha$,$\beta$ are the zeroes of a polynomial $p(x)=x^2+x-1$, then $\frac{1}{\alpha}+\frac{1}{\beta}$ equals to 
\begin{enumerate}
\item $1$
\item $2$
\item $-1$
\item $\frac{-1}{2}$
\end{enumerate}



\end{enumerate}
